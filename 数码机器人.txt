local function updateCharacterRefs()
    local success, err = pcall(function()
        local char = getCharacter()
        if char then
            character.humanoid = char:FindFirstChildOfClass("Humanoid")
            character.rootPart = char:FindFirstChild("HumanoidRootPart")
            if character.humanoid and not character.lastHealth then
                character.lastHealth = character.humanoid.Health
            end
        end
    end)
    if not success then
        warn("Failed to update character references:", err)
    end
end

local function output(msg, success)
    local text = "[System]: " .. msg .. (success and " ✔" or "")
    if print then print(text) end

        local char = getCharacter()
        if char and char:FindFirstChild("Head") then
            game:GetService("Chat"):Chat(char.Head, text)
        end
    end)
    if not success then
        warn("Failed to send chat message:", err)
    end
end

-- ========== Prediction ==========
local function updateTargetVelocity()
    if not character.prediction_enabled then return end
    if not character.follow_target then return end
    
    local targetPlayer = players:FindFirstChild(character.follow_target)
    if not targetPlayer or not targetPlayer.Character then return end
    
    local targetRoot = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not targetRoot then return end
    
    local currentPos = targetRoot.Position
    local currentTime = tick()
    
    if character.target_last_pos and character.target_last_time then
        local deltaTime = currentTime - character.target_last_time
        if deltaTime > 0 and deltaTime < 0.5 then
            local velocity = (currentPos - character.target_last_pos) / deltaTime
            if velocity.Magnitude < 500 then
                character.target_velocity = velocity
            end
        end
    end
    
    character.target_last_pos = currentPos
    character.target_last_time = currentTime
end

local function getPredictedPosition(predictionTime)
    if not character.prediction_enabled then
        return nil
    end
    
    local targetPlayer = players:FindFirstChild(character.follow_target)
    if not targetPlayer or not targetPlayer.Character then return nil end
    
    local targetRoot = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not targetRoot then return nil end
    
    local currentPos = targetRoot.Position
    local velocity = character.target_velocity
    
    if velocity.Magnitude < 1 then
        return currentPos
    end
    
    local predictedPos = currentPos + velocity * predictionTime
    local maxDistance = 50
    if (predictedPos - currentPos).Magnitude > maxDistance then
        predictedPos = currentPos + (velocity.Unit * maxDistance)
    end
    
    return predictedPos
end

local function togglePrediction(state)
    if state == nil then
        character.prediction_enabled = not character.prediction_enabled
    else
        character.prediction_enabled = state
    end
    
    if character.prediction_enabled then
        character.target_velocity = Vector3.new(0, 0, 0)
        character.target_last_pos = nil
        character.target_last_time = nil
        output("Prediction Enabled - Will predict target movement", true)
    else
        output("Prediction Disabled", true)
    end
    updateUI()
end

    lastPlayerList = {}
    for i, p in ipairs(playersList) do
        if p ~= player then

        end

end

local function getPlayerNameByIndex(index)
    updatePlayerList()
    if index < 1 or index > #lastPlayerList then return nil end
    return lastPlayerList[index]
end

-- ========== Send Message ==========
local function sendChatMessage(message, channel)
    channel = channel or "All"
    local success = false

    local TextChatService = game:GetService("TextChatService")
    if TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
        local textChannel = TextChatService.TextChannels:FindFirstChild("RBXGeneral")
        if textChannel then
            pcall(function() textChannel:SendAsync(message) end)
        end
    else
        local sayRequest = game:GetService("ReplicatedStorage"):FindFirstChild("DefaultChatSystemChatEvents")
        if sayRequest then
            sayRequest = sayRequest:FindFirstChild("SayMessageRequest")
            if sayRequest then
                pcall(function() sayRequest:FireServer(message, channel) end)
            end
        end
    end
end

-- ========== Infinite Jump ==========
local function toggleInfiniteJump()
    character.infinite_jump = not character.infinite_jump
    if character.infinite_jump then
        if jumpConnection then jumpConnection:Disconnect() end
        jumpConnection = userInputService.JumpRequest:Connect(function()
            if character.infinite_jump and character.humanoid then
                character.humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
            end

        if jumpConnection then
            jumpConnection:Disconnect()

        end

-- ========== Noclip (Fixed - Permanent, Never Interrupted) ==========
local function maintainImmunity()
    if not character.immunity then return end
    
    local char = getCharacter()
    if not char then return end
    
    local parts = char:GetDescendants()
    for _, part in ipairs(parts) do
        if part:IsA("BasePart") then
            part.CanCollide = false
            part.Transparency = 0.5
        end
    end
    
    local rootPart = char:FindFirstChild("HumanoidRootPart")
    if rootPart then
        character.rootPart = rootPart
        rootPart.CanCollide = false
        rootPart.Transparency = 0.5
    end
end

local function applyImmunityToAllParts()
    local char = getCharacter()
    if not char then return end
    
    local parts = char:GetDescendants()
    for _, part in ipairs(parts) do
        if part:IsA("BasePart") then
            part.CanCollide = false
            part.Transparency = 0.5
        end
    end
    
    local rootPart = char:FindFirstChild("HumanoidRootPart")
    if rootPart then
        character.rootPart = rootPart
        rootPart.CanCollide = false
        rootPart.Transparency = 0.5
    end
end

local function restorePartsCollision()
    local char = getCharacter()
    if not char then return end
    
    for _, part in ipairs(char:GetDescendants()) do
        if part:IsA("BasePart") then
            part.CanCollide = true
            part.Transparency = 0
        end
    end
    
    local rootPart = char:FindFirstChild("HumanoidRootPart")
    if rootPart then
        character.rootPart = rootPart
        rootPart.CanCollide = true
        rootPart.Transparency = 0
    end
end

local function onCharacterAddedForImmunity(newChar)
    if character.immunity then
        task.wait(0.3)
        applyImmunityToAllParts()
        
        if not immunityMaintainConnection then
            immunityMaintainConnection = runService.Heartbeat:Connect(maintainImmunity)
        end
    end
end

local function setImmunity(enabled)
    if enabled then
        if character.immunity then
            if immunityMaintainConnection then
                immunityMaintainConnection:Disconnect()
                immunityMaintainConnection = nil
            end
        end
        
        applyImmunityToAllParts()
        immunityMaintainConnection = runService.Heartbeat:Connect(maintainImmunity)
        
        if not character.characterAddedConnection then
            character.characterAddedConnection = player.CharacterAdded:Connect(onCharacterAddedForImmunity)
        end
        
        character.immunity = true
        character.wallHackActive = true
        output("Noclip Enabled (Permanent, Never Interrupted)", true)
    else
        if immunityMaintainConnection then
            immunityMaintainConnection:Disconnect()
            immunityMaintainConnection = nil
        end
        
        restorePartsCollision()
        character.immunity = false
        character.wallHackActive = false
        output("Noclip Disabled", true)
    end
    updateUI()
end

-- ========== Spin Control ==========
local function toggleSpin(enabled)
    if enabled == nil then
        character.spin_enabled = not character.spin_enabled
    else
        character.spin_enabled = enabled
    end
    local char = getCharacter()
    if not char then return end
    local root = char:FindFirstChild("HumanoidRootPart")
    if not root then return end
    local humanoid = char:FindFirstChildOfClass("Humanoid")
    if humanoid then
        humanoid.AutoRotate = not character.spin_enabled
    end
    if character.spin_enabled then
        if character.spin_angularVelocity then
            pcall(function() character.spin_angularVelocity:Destroy() end)
            character.spin_angularVelocity = nil
        end
        local av = Instance.new("AngularVelocity")
        local att = root:FindFirstChild("RootAttachment") or root:FindFirstChild("RootRigAttachment")
        if att then
            av.Attachment0 = att
        end
        av.MaxTorque = math.huge
        av.AngularVelocity = Vector3.new(0, character.spin_speed, 0)
        av.Parent = root
        character.spin_angularVelocity = av
        if enabled == nil then output("Spin Enabled, Speed "..character.spin_speed, true) end
    else
        if character.spin_angularVelocity then
            pcall(function() character.spin_angularVelocity:Destroy() end)
            character.spin_angularVelocity = nil
        end
        if enabled == nil then output("Spin Disabled", true) end
    end
    updateUI()
end

local function setSpinSpeed(speed)
    local s = tonumber(speed)
    if not s then
        output("Please enter a valid number", false)
        return
    end
    character.spin_speed = s
    if character.spin_enabled and character.spin_angularVelocity then
        character.spin_angularVelocity.AngularVelocity = Vector3.new(0, s, 0)
    end
    output("Spin speed set to " .. s, true)
    updateUI()
end

-- ========== Float Control ==========
local function setFloating(enabled)
    updateCharacterRefs()
    if not character.rootPart then output("Cannot get character position", false) return end
    if enabled and not character.floating then
        if character.floatBody then pcall(function() character.floatBody:Destroy() end); character.floatBody = nil end
        local body = Instance.new("BodyVelocity")
        body.MaxForce = Vector3.new(0, math.huge, 0)
        body.Velocity = Vector3.new(0, 0, 0)
        body.P = 10000
        body.Parent = character.rootPart
        character.floatBody = body
        character.floating = true
        output("Float Enabled", true)
    elseif not enabled and character.floating then
        if character.floatBody then pcall(function() character.floatBody:Destroy() end); character.floatBody = nil end
        character.floating = false
        output("Float Disabled", true)
    end
    updateUI()
end

-- ========== Damage Immunity (Other Players) ==========
local function setDamageImmunityForPlayer(targetName, enable)
    local targetPlayer = players:FindFirstChild(targetName)
    if not targetPlayer then output("Player "..targetName.." is offline", false) return end
    if enable then
        if damageImmunityList[targetName] then task.cancel(damageImmunityList[targetName]); damageImmunityList[targetName] = nil end
        local thread = task.spawn(function()
            while true do
                if not character or not damageImmunityList[targetName] then break end
                local p = players:FindFirstChild(targetName)
                if p and p.Character and p.Character:FindFirstChildOfClass("Humanoid") then
                    local hum = p.Character:FindFirstChildOfClass("Humanoid")
                    if hum and hum.Health < hum.MaxHealth then
                        hum.Health = hum.MaxHealth
                    end
                else
                    output("Player "..targetName.." disappeared, immunity auto-disabled", true)
                    damageImmunityList[targetName] = nil
                    break
                end
                task.wait(0.1)
            end
        end)
        damageImmunityList[targetName] = thread
        output("Damage immunity enabled for "..targetName, true)
    else
        if damageImmunityList[targetName] then task.cancel(damageImmunityList[targetName]); damageImmunityList[targetName] = nil end
        output("Damage immunity disabled for "..targetName, true)
    end
    updateUI()
end

local function setCharacterHeight(y)
    local height = tonumber(y)
    if not height then output("Please enter a valid number", false) return end
    if height < MIN_HEIGHT or height > MAX_HEIGHT then output("Height out of range ("..MIN_HEIGHT.."~"..MAX_HEIGHT..")", false) return end
    updateCharacterRefs()
    if character.rootPart then
        local pos = character.rootPart.Position
        character.rootPart.CFrame = CFrame.new(pos.X, height, pos.Z)
        if not character.floating then setFloating(true) end
        output("Height set to "..height, true)
        updateUI()
        if ui.heightInput and ui.heightInput.Parent then ui.heightInput.Text = "" end
    else
        output("Cannot get character position", false)
    end
end

-- ========== Get Player ==========
local function getNearestPlayerName()
    updateCharacterRefs()
    if not character.rootPart then return nil end
    local nearest, minDist = nil, math.huge
    for _, p in ipairs(players:GetPlayers()) do
        if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            local dist = (p.Character.HumanoidRootPart.Position - character.rootPart.Position).Magnitude
            if dist < minDist then minDist = dist; nearest = p.Name end
        end
    end
    return nearest
end

local function smoothMoveTo(targetPosition)
    if not character or not character.rootPart then return end
    local currentPos = character.rootPart.Position
    local distance = (targetPosition - currentPos).Magnitude
    if distance < 3 then
        character.rootPart.CFrame = CFrame.new(targetPosition)
        character.rootPart.Velocity = Vector3.new(0,0,0)
        return
    end
    local direction = (targetPosition - currentPos).Unit
    local moveSpeed = math.min(character.speed, distance * 2)
    character.rootPart.Velocity = direction * moveSpeed
    if math.abs(targetPosition.Y - currentPos.Y) > 0.5 then
        setFloating(true)
    end
end

local function smoothMoveHorizontal(targetPosition)
    if not character or not character.rootPart then return end
    local currentPos = character.rootPart.Position
    local targetXZ = Vector3.new(targetPosition.X, currentPos.Y, targetPosition.Z)
    local distance = (targetXZ - currentPos).Magnitude
    if distance < 3 then
        character.rootPart.CFrame = CFrame.new(targetXZ)
        character.rootPart.Velocity = Vector3.new(0,0,0)
        return
    end
    local direction = (targetXZ - currentPos).Unit
    local moveSpeed = math.min(character.speed, distance * 2)
    character.rootPart.Velocity = direction * moveSpeed
end

local function teleportToPlayer(targetName, smooth)
    local targetPlayer = players:FindFirstChild(targetName)
    if not targetPlayer or not targetPlayer.Character or not targetPlayer.Character:FindFirstChild("HumanoidRootPart") then
        output("Player "..targetName.." is offline or has no character", false)
        return
    end
    updateCharacterRefs()
    if not character.rootPart then
        output("Cannot get your character position", false)
        return
    end
    local targetPart = targetPlayer.Character.HumanoidRootPart
    local targetPos = targetPart.Position
    local dest = Vector3.new(targetPos.X, targetPos.Y, targetPos.Z)
    if smooth then
        smoothMoveTo(dest)
        output("Smooth moving to "..targetPlayer.Name, true)
    else
        character.rootPart.CFrame = CFrame.new(dest)
        character.rootPart.Velocity = Vector3.new(0,0,0)
        if not character.floating then setFloating(true) end
        output("Teleported to "..targetPlayer.Name.."'s position", true)
    end
    if ui.tpInput and ui.tpInput.Parent then
        ui.tpInput.Text = ""
    end
end

local function teleportHorizontal(targetPosition)
    if not character.rootPart then return end
    local currentPos = character.rootPart.Position
    character.rootPart.CFrame = CFrame.new(targetPosition.X, currentPos.Y, targetPosition.Z)
    character.rootPart.Velocity = Vector3.new(0,0,0)
    if not character.floating then setFloating(true) end
end

local function teleportToTarget()
    local target = getTargetPlayer()
    if not target or not target.Character or not target.Character:FindFirstChild("HumanoidRootPart") then

        return

    notify("DIGITAL", "Teleported to "..target.Name, 3)
end

            while character.loop_teleport_to do

                task.wait(0.1)
            end
        end)
        notify("DIGITAL", "Loop teleport to target enabled", 3)
    else
        notify("DIGITAL", "Loop teleport to target disabled", 3)
    end
end

local function flingTargetOnce()
    local target = getTargetPlayer()
    if not target then
        notify("DIGITAL", "Please select a target first", 3)
        return
    end
    local localChar = getCharacter()
    if not localChar then return end
    local localHum = localChar:FindFirstChildOfClass("Humanoid")
    local localRoot = localHum and localHum.RootPart
    if not localRoot then return end

    local targetChar = target.Character
    if not targetChar then
        notify("DIGITAL", "Target character does not exist", 3)
        return
    end
    local targetHum = targetChar:FindFirstChildOfClass("Humanoid")
    local targetRoot = targetHum and targetHum.RootPart
    local targetHead = targetChar:FindFirstChild("Head")
    if not targetRoot and not targetHead then
        notify("DIGITAL", "Target has no flingable part", 3)
        return
    end

    getgenv().OldPos = localRoot.CFrame
    getgenv().FPDH = workspace.FallenPartsDestroyHeight
    workspace.FallenPartsDestroyHeight = 0/0

    if targetHead then
        workspace.CurrentCamera.CameraSubject = targetHead
    elseif targetRoot then
        workspace.CurrentCamera.CameraSubject = targetHum
    end

    localHum:SetStateEnabled(Enum.HumanoidStateType.Seated, false)

    local bv = Instance.new("BodyVelocity")
    bv.Name = "FlingVelocity"
    bv.Velocity = Vector3.new(9e7, 9e8, 9e7)
    bv.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    bv.Parent = localRoot

    local targetPart = targetHead or targetRoot
    local startTime = tick()
    local timeout = 3

    while localRoot and targetPart and targetPart.Parent do
        local dt = tick() - startTime
        if dt > timeout then break end
        local angle = dt * 500
        local offset = CFrame.new(0, 1.5, 0) * CFrame.Angles(0, math.rad(angle), 0)
        localRoot.CFrame = CFrame.new(targetPart.Position) * offset
        localChar:SetPrimaryPartCFrame(CFrame.new(targetPart.Position) * offset)
        task.wait()
    end

    bv:Destroy()
    localHum:SetStateEnabled(Enum.HumanoidStateType.Seated, true)
    workspace.CurrentCamera.CameraSubject = localHum

    repeat
        localRoot.CFrame = getgenv().OldPos * CFrame.new(0, 0.5, 0)
        localChar:SetPrimaryPartCFrame(getgenv().OldPos * CFrame.new(0, 0.5, 0))
        localHum:ChangeState("GettingUp")
        for _, part in ipairs(localChar:GetDescendants()) do
            if part:IsA("BasePart") then
                part.Velocity = Vector3.new()
                part.RotVelocity = Vector3.new()
            end
        end
        task.wait()
    until (localRoot.Position - getgenv().OldPos.p).Magnitude < 25

    workspace.FallenPartsDestroyHeight = getgenv().FPDH
    notify("DIGITAL", "Fling completed", 3)
end

-- ========== Hitbox Feature (GUST Style - Modify Collision Box) ==========
local function updateHitboxes()
    if not character.hitbox_enabled then
        -- When hitbox is off, restore all players' RootPart to original state
        for _, p in ipairs(players:GetPlayers()) do
            if p ~= player then
                local char = p.Character
                if char then
                    local root = char:FindFirstChild("HumanoidRootPart")
                    if root then
                        root.Size = Vector3.new(2, 2, 1)
                        root.Transparency = 0
                        root.BrickColor = BrickColor.new("Medium stone grey")
                        root.Material = Enum.Material.Plastic
                        root.CanCollide = true
                    end
                end
            end
        end
        return
    end

    -- When hitbox is on, modify other players' RootPart
    for _, p in ipairs(players:GetPlayers()) do
        if p ~= player then
            -- Team check
            if character.hitbox_teamCheck and player.Team == p.Team then
                -- Same team, skip
            else
                local char = p.Character
                if char then
                    local root = char:FindFirstChild("HumanoidRootPart")
                    if root then
                        -- Immune players are restored
                        if hitboxImmunePlayers[p.Name] then
                            root.Size = Vector3.new(2, 2, 1)
                            root.Transparency = 0
                            root.BrickColor = BrickColor.new("Medium stone grey")
                            root.Material = Enum.Material.Plastic
                            root.CanCollide = true
                        else
                            -- Modify hitbox size, transparency, material, color, collision
                            root.Size = Vector3.new(character.hitbox_size, character.hitbox_size, character.hitbox_size)
                            root.Transparency = character.hitbox_transparency
                            root.BrickColor = character.hitbox_color
                            root.Material = Enum.Material.Neon
                            root.CanCollide = false
                        end
                    end
                end
            end
        end
    end
end

local function toggleHitbox(enabled)
    if enabled == nil then
        character.hitbox_enabled = not character.hitbox_enabled
    else
        character.hitbox_enabled = enabled
    end

    if character.hitbox_enabled then
        if character.hitbox_connection then
            character.hitbox_connection:Disconnect()
        end
        character.hitbox_connection = runService.RenderStepped:Connect(updateHitboxes)
        output("Hitbox Enabled, Size " .. character.hitbox_size, true)
    else
        if character.hitbox_connection then
            character.hitbox_connection:Disconnect()
            character.hitbox_connection = nil
            updateHitboxes()  -- Restore all targets immediately
        end
        output("Hitbox Disabled", true)
    end
    updateUI()
end

-- ========== Ultra Mode (God Mode) ==========
local function toggleUltra(state)
    if state == nil then
        character.ultra_enabled = not character.ultra_enabled
    else
        character.ultra_enabled = state
    end

    if character.ultra_enabled then
        if character.ultra_thread then task.cancel(character.ultra_thread) end

        if not character.follow_target then
            output("Please set a follow target first (.f <playername>)", false)
            character.ultra_enabled = false
            return
        end

        character.ultra_thread = task.spawn(function()
            local angle = 0
            local distance = 14
            local speed = 0.1
            
            while character.ultra_enabled do
                local targetName = character.follow_target
                if not targetName then
                    output("Target lost, stopping Ultra Mode", true)
                    character.ultra_enabled = false
                    break
                end

                local targetPlayer = players:FindFirstChild(targetName)
                if not targetPlayer or not targetPlayer.Character then
                    output("Target character not found, stopping Ultra Mode", true)
                    character.ultra_enabled = false
                    break
                end

                local targetHum = targetPlayer.Character:FindFirstChildOfClass("Humanoid")
                local targetPart = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
                
                if not targetHum or targetHum.Health <= 0 or not targetPart then
                    output("Target dead or invalid, stopping Ultra Mode", true)
                    character.ultra_enabled = false
                    break
                end

                updateCharacterRefs()
                if not character.rootPart then
                    task.wait(0.5)
                    continue
                end

                local targetPos = targetPart.Position

                local offsetX = math.cos(angle) * distance
                local offsetZ = math.sin(angle) * distance
                local newPos = Vector3.new(
                    targetPos.X + offsetX,
                    targetPos.Y + 2,
                    targetPos.Z + offsetZ
                )

                character.rootPart.CFrame = CFrame.lookAt(newPos, targetPos)
                character.rootPart.Velocity = Vector3.new(0,0,0)
                if not character.floating then setFloating(true) end

                local cam = workspace.CurrentCamera
                cam.CameraType = Enum.CameraType.Scriptable
                local camDistance = 20
                local camHeight = 8
                local camPos = targetPos + Vector3.new(math.cos(angle) * camDistance, camHeight, math.sin(angle) * camDistance)
                cam.CFrame = CFrame.lookAt(camPos, targetPos)

                angle = angle + speed

                if angle % (math.pi) < speed then
                    distance = (distance == 14) and 20 or 14
                end

                if angle > 2 * math.pi then
                    angle = angle - 2 * math.pi
                end

                task.wait(0.05)
            end

            updateCharacterRefs()
            if character.humanoid then
                workspace.CurrentCamera.CameraSubject = character.humanoid
                workspace.CurrentCamera.CameraType = Enum.CameraType.Custom
            end
        end)
        output("Ultra Mode Enabled: God camera orbiting target", true)
    else
        if character.ultra_thread then
            task.cancel(character.ultra_thread)
            character.ultra_thread = nil
        end
        updateCharacterRefs()
        if character.humanoid then
            workspace.CurrentCamera.CameraSubject = character.humanoid
            workspace.CurrentCamera.CameraType = Enum.CameraType.Custom
        end
        output("Ultra Mode Disabled", true)
    end
    updateUI()
end

-- ========== Follow ==========
local function stopFollow()
    if character.follow_target then
        character.follow_target = nil
        if followLoop then task.cancel(followLoop); followLoop = nil end
        output("Stopped following", true)
        updateUI()
    end
end

local function startFollow(targetName)
    if character.follow_target then stopFollow() end
    local targetPlayer = players:FindFirstChild(targetName)
    if not targetPlayer then output("Player "..targetName.." is offline, waiting...", true) end
    character.follow_target = targetName
    if followLoop then task.cancel(followLoop); followLoop = nil end
    followLoop = task.spawn(function()
        local waitTimer = 0
        while character and character.follow_target do
            local target = players:FindFirstChild(character.follow_target)
            if target and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
                updateCharacterRefs()
                if character and character.rootPart then
                    local targetChar = target.Character
                    local targetPart = targetChar:FindFirstChild("HumanoidRootPart")
                    local targetHead = targetChar:FindFirstChild("Head")
                    local targetBaseY = targetHead and targetHead.Position.Y or (targetPart.Position.Y + 2.5)
                    local targetY = targetBaseY + character.follow_height_offset
                    local targetPos = targetPart.Position
                    
                    if character.prediction_enabled then
                        local predictedPos = getPredictedPosition(0.2)
                        if predictedPos then
                            targetPos = predictedPos
                        end
                    end

                    if character.smooth_follow then
                        local currentPos = character.rootPart.Position
                        character.rootPart.CFrame = CFrame.new(currentPos.X, targetY, currentPos.Z)
                        smoothMoveHorizontal(Vector3.new(targetPos.X, targetY, targetPos.Z))
                    else
                        character.rootPart.CFrame = CFrame.new(targetPos.X, targetY, targetPos.Z)
                        character.rootPart.Velocity = Vector3.new(0,0,0)
                        if not character.floating then setFloating(true) end
                    end
                end
                waitTimer = 0
                task.wait(character.follow_interval)
            else
                waitTimer = waitTimer + 1
                if waitTimer > 30 then waitTimer = 30 end
                task.wait(0.1 * waitTimer)
            end
        end
    end)
    output("Following "..targetName.." (Head offset +"..character.follow_height_offset..")", true)
    sendChatMessage("☠️HUNTING " .. targetName .. "!!!☠️", "All")
    updateUI()
end

            {ui.predictionBtn, character.prediction_enabled and "Prediction OFF" or "Prediction ON"},
        }
        for _, update in ipairs(btnUpdates) do
            local btn, text = update[1], update[2]
            if btn and btn.Parent then btn.Text = text end
        end
        if ui.heightInput and ui.heightInput.Parent and character.rootPart then
            ui.heightInput.PlaceholderText = tostring(math.floor(character.rootPart.Position.Y))
        end
        if ui.fhInput and ui.fhInput.Parent then
            ui.fhInput.PlaceholderText = "Current:" .. character.follow_height_offset
        end
        if ui.yLockInput and ui.yLockInput.Parent then
            ui.yLockInput.PlaceholderText = "Current:" .. character.y_lock_value
        end
    end)
    if not success then warn("UI update failed:", err) end
end

        ui.loadingText = nil
    end

-- ========== Command Parser ==========
local function executeCommand(cmd)
    cmd = cmd:gsub("^%s+", ""):gsub("%s+$", "")
    
    if not bot_active then
        if cmd == PASSWORD then
            bot_active = true
            output("DIGITAL Online", true)
            output("Type '.l' to see commands", true)
            sendChatMessage("DIGITAL KILL MODE ACTIVATED!!! Made by etern_ncz(concealed)👁️", "All")
            updateUI()
            sendChatMessage("I will always be watching you.......👁️", "All")
            -- Auto enable default features
            setImmunity(true)
            toggleInfiniteJump()
            character.hitbox_size = 1200
            toggleHitbox(true)
            output("⚡ Auto enabled: Noclip, Infinite Jump, Hitbox (Size 1200)", true)
        else
            output("Wrong password", false)
        end
        return
    end
    
    if cmd == "" then return end

    if cmd:sub(1,1) == "." then
        local dotcmd, arg = cmd:match("^%.([^%s]+)%s*(.*)$")
        if not dotcmd then
            dotcmd = cmd:match("^%.([^%s]+)$")
            arg = ""
        end
        if not dotcmd then return end
        dotcmd = dotcmd:lower()

        if dotcmd == "pred" or dotcmd == "prediction" then
            togglePrediction()
        elseif dotcmd == "w" then
            setImmunity(not character.immunity)
        elseif dotcmd == "j" then
            toggleInfiniteJump()
        elseif dotcmd == "s" then
            adjustSpeed()
        elseif dotcmd == "r" then
            resetCharacter()
        elseif dotcmd == "l" then
            output("Single letter: .w .j .s .r .l .p .f .t .b .m1 .m2 .c .as .pred", true)
        elseif dotcmd == "p" then
            refreshPlayerList()
        elseif dotcmd == "f" then
            if arg and arg ~= "" then
                local index = tonumber(arg)
                if index then
                    local name = getPlayerNameByIndex(index)
                    if name then startFollow(name) else output("Invalid number", false) end
                else
                    startFollow(arg)
                end
            else
                followNearest()
            end
        elseif dotcmd == "t" then
            if arg and arg ~= "" then
                local index = tonumber(arg)
                if index then
                    local name = getPlayerNameByIndex(index)
                    if name then teleportToPlayer(name, character.smooth_follow) end
                else
                    teleportToPlayer(arg, character.smooth_follow)
                end
            else
                local nearest = getNearestPlayerName()
                if nearest then teleportToPlayer(nearest, character.smooth_follow) end
            end
        elseif dotcmd == "b" then
            toggleBlock()
        elseif dotcmd == "m1" then
            toggleM1Loop()
        elseif dotcmd == "m2" then
            toggleM2Loop()
        elseif dotcmd == "c" then
            local any_on = character.block_enabled or character.m1_enabled or character.m2_enabled
            if any_on then
                if character.block_enabled then toggleBlock(false) end
                if character.m1_enabled then toggleM1Loop(false) end
                if character.m2_enabled then toggleM2Loop(false) end
                output("All attack loops disabled", true)
            else
                if not character.block_enabled then toggleBlock(true) end
                if not character.m1_enabled then toggleM1Loop(true) end
                if not character.m2_enabled then toggleM2Loop(true) end
                output("All attack loops enabled", true)
            end
        elseif dotcmd == "as" then
            toggleAutoSuicide()
        elseif dotcmd == "eternal" then
            if arg and arg ~= "" then
                local index = tonumber(arg)
                local targetName
                if index then
                    targetName = getPlayerNameByIndex(index)
                    if not targetName then
                        output("Invalid number", false)
                        return
                    end
                else
                    targetName = arg
                end
                local targetPlayer = players:FindFirstChild(targetName)
                if targetPlayer then
                    hitboxImmunePlayers[targetPlayer.Name] = true
                    sendChatMessage(targetPlayer.Name .. " IMMUNE!!!", "All")
                    output("Hitbox immunity enabled for " .. targetPlayer.Name, true)
                else
                    output("Player " .. targetName .. " is offline", false)
                end
            else
                output("Usage: .eternal <player number>", false)
            end
        elseif dotcmd == "bcd" then
            if arg and arg ~= "" then
                setBlockCooldown(arg)
            else
                output("Usage: .bcd <seconds>", false)
            end
        elseif dotcmd == "fh" then
            if arg and arg ~= "" then
                setFollowHeightOffset(arg)
            else
                output("Usage: .fh <offset>", false)
            end
        elseif dotcmd == "yl" then
            toggleYLock()
        elseif dotcmd == "ylock" then
            if arg and arg ~= "" then
                setYLockValue(arg)
            else
                output("Usage: .ylock <height>", false)
            end
        elseif dotcmd == "ultra" then
            toggleUltra()
        elseif dotcmd == "players" or dotcmd == "list" then
            refreshPlayerList()
        elseif dotcmd == "y" then
            if arg and arg ~= "" then
                setCharacterHeight(arg)
            else
                output("Usage: .y <height>", false)
            end
        elseif dotcmd == "tp" or dotcmd == "goto" then
            local targetName = (ui.targetInput and ui.targetInput.Text ~= "") and ui.targetInput.Text or arg
            if targetName and targetName ~= "" then 
                teleportToPlayer(targetName, character.smooth_follow) 
            else 
                output("Usage: .tp <playername>", false) 
            end
        elseif dotcmd == "fl" or dotcmd == "fly" then 
            setFloating(not character.floating)
        elseif dotcmd == "follow" then
            if arg and arg ~= "" then
                if arg == "stop" then 
                    stopFollow() 
                else 
                    startFollow(arg) 
                end
            else 
                output("Usage: .follow <playername> or .follow stop", false) 
            end
        elseif dotcmd == "mode" then
            toggleFollowMode()
        elseif dotcmd == "di" or dotcmd == "dmg" then
            if arg == "" or arg == "me" then 
                output("Self immunity feature removed", false)
            elseif arg == "list" then
                local list = {}
                for name,_ in pairs(damageImmunityList) do 
                    table.insert(list, name) 
                end
                output("Immunity players: " .. table.concat(list, ", "), true)
            else
                if damageImmunityList[arg] then 
                    setDamageImmunityForPlayer(arg, false)
                else 
                    setDamageImmunityForPlayer(arg, true) 
                end
            end
        elseif dotcmd == "wall" then
            setImmunity(not character.immunity)
        elseif dotcmd == "spin" then
            if arg == "" then
                toggleSpin()
            elseif arg:match("^%d+$") then
                setSpinSpeed(tonumber(arg))
            else
                output("Usage: .spin or .spin <speed>", false)
            end
        elseif dotcmd == "hitbox" then
            if arg == "" then
                toggleHitbox(not character.hitbox_enabled)
            elseif arg:match("^%d+$") then
                setHitboxSize(tonumber(arg))
            else
                output("Usage: .hitbox or .hitbox <size>", false)
            end
        elseif dotcmd == "teamcheck" then
            toggleTeamCheck()
        elseif dotcmd == "fling" then
            if arg == "" then
                flingTargetOnce()
            else
                output("Usage: .fling (no parameters, flings current target)", false)
            end
        elseif dotcmd == "suck" then
            toggleSuckAll(not character.loop_suck_all)
        elseif dotcmd == "view" then
            toggleViewPlayer(not character.viewing)
        elseif dotcmd == "aim" then
            toggleAimbot(not character.aimbot)
        else 
            output("Unknown dot command: ."..dotcmd, false) 
        end
        return
    end

    local lcmd = cmd:lower()
    if lcmd == "idle" then 
        character.attacking = false
        output("Idle state", true)
        updateUI()
    elseif lcmd == "reset" then 
        resetCharacter()
    elseif lcmd == "speed" then 
        adjustSpeed()
    elseif lcmd == "list" then
        output("Single letter commands: .w .j .s .r .l .p .f .t .b .m1 .m2 .c .as .pred", true)
        output("Other commands: .tp .fl .follow .mode .di .spin .hitbox .teamcheck .fling .suck .view .aim .bcd .eternal .fh .yl .ylock .ultra", true)
    else 
        output("Unknown command", false) 
    end
end

-- ========== Main Loop ==========
local function run()
    if heartbeatConnection then heartbeatConnection:Disconnect(); heartbeatConnection = nil end
    
    showLoadingScreen()
    
    createUI()
    createMenuButton()

    player.CharacterAdded:Connect(function()
        task.wait(0.5)
        updateCharacterRefs()
        if character.floatBody then pcall(function() character.floatBody:Destroy() end); character.floatBody = nil end
        character.floating = false
        if character.wallHackActive then
            task.wait(0.2)
            setImmunity(true)
        end
        if character.spin_enabled then toggleSpin(true) end
        if character.block_enabled then toggleBlock(true) end
        if character.m1_enabled then toggleM1Loop(true) end
        if character.m2_enabled then toggleM2Loop(true) end
        updateUI()
    end)

    heartbeatConnection = runService.Heartbeat:Connect(function()
        if bot_active and character then
            updateUI()
            maintainSpeed()
            updateTargetVelocity()

            if character.y_lock_enabled and character.rootPart then
                local pos = character.rootPart.Position
                if math.abs(pos.Y - character.y_lock_value) > 0.1 then
                    character.rootPart.CFrame = CFrame.new(pos.X, character.y_lock_value, pos.Z)
                    if not character.floating then setFloating(true) end
                end
            end

            if character.auto_suicide and character.humanoid and character.humanoid.Health <= 5 then
                task.wait(0.1)
                local char = getCharacter()
                if char then
                    char:BreakJoints()
                end
            end
        end
    end)

    player.Chatted:Connect(executeCommand)
    output("Script loaded, enter password 'Tyrannosaurus rex activated' to activate", true)
    output("Noclip never interrupted, auto-reapplies on respawn", true)
    output("New features: .pred prediction toggle | .ultra orbit target", true)
    output("Hitbox now modifies target collision box size (visual range), no longer sends attack events", true)
end

local success, err = pcall(run)
if not success then
    warn("Script startup failed:", err)
end
